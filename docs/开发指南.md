# SPK Workers - 开发指南

## 开发环境设置

### 1. 安装依赖

```bash
# 克隆项目
git clone https://github.com/your-repo/spk-workers.git
cd spk-workers

# 安装 Node.js 依赖
npm install
```

### Wrangler 配置

复制配置文件模板：

```bash
cp examples/wrangler.toml.example wrangler.toml
```

编辑 `wrangler.toml`，填入您的实际配置值。详细说明请参考 [examples/README.md](../examples/README.md)。

### 3. 环境变量

复制环境变量模板用于本地开发：

```bash
cp examples/.dev.vars.example .dev.vars
# 编辑 .dev.vars 填入实际值
```

**注意**: `SSPKS_API_KEY` 用于保护上传接口，请使用强密码并妥善保管。

### 4. 启动开发服务器

```bash
npm run dev
```

访问 `http://localhost:8787` 查看应用。

## 代码规范

### TypeScript 规范

- 使用 **ES2022** 目标
- 启用严格模式 (`strict: true`)
- 使用 **2 空格**缩进
- 使用 **单引号** 字符串
- **末尾分号** 可选

### 命名规范

| 类型 | 规范 | 示例 |
|------|------|------|
| 类 | PascalCase | `PackageFilter` |
| 接口 | PascalCase | `PackageMetadata` |
| 方法 | camelCase | `getFilteredPackageList` |
| 常量 | SCREAMING_SNAKE | `MAX_CACHE_AGE` |
| 文件 | kebab-case | `package-filter.ts` |
| 类型 | PascalCase | `ArchFilter` |

### 类型定义

```typescript
// Good
interface PackageMetadata {
  package: string;
  version: string;
  arch: string[];
}

// Avoid
type PackageMetadata = {
  package: string;
  version: string;
  arch: string[];
};
```

### 异步处理

```typescript
// 优先使用 async/await
async function fetchPackage(name: string): Promise<Package> {
  const obj = await env.SPKS_BUCKET.get(`packages/${name}.spk`);
  if (!obj) throw new Error('Package not found');
  return Package.fromR2(env.SPKS_BUCKET, name);
}

// 避免不必要的 Promise 包装
// Bad
function fetchPackage(name: string): Promise<Package> {
  return new Promise(async (resolve) => {
    const obj = await env.SPKS_BUCKET.get(`packages/${name}.spk`);
    resolve(obj);
  });
}
```

### 3. 处理器模块 (Handler)

位置: `src/handlers/*.ts`

```typescript
export interface HandlerInterface {
  canHandle(request: Request): boolean;
  handle(request: Request, env: Env, ctx: ExecutionContext): Promise<Response>;
}

export abstract class AbstractHandler implements HandlerInterface {
  abstract canHandle(request: Request): boolean;
  abstract handle(request: Request, env: Env, ctx: ExecutionContext): Promise<Response>;

  protected json(data: unknown, init?: ResponseInit): Response {
    return Response.json(data, {
      ...init,
      headers: { "Content-Type": "application/json", ...init?.headers },
    });
  }

  protected redirect(url: string, status: number = 302): Response {
    return new Response(null, { status, headers: { Location: url } });
  }
}

export class SynologyHandler extends AbstractHandler {
  canHandle(request: Request): boolean {
    const url = new URL(request.url);
    return url.pathname === "/" || url.pathname === "/api/info";
  }
}
```

### 4. 上传处理器模块 (UploadHandler)

位置: `src/handlers/UploadHandler.ts`

```typescript
import { AbstractHandler } from "./AbstractHandler";
import { Config } from "../config/Config";
import { Package } from "../package/Package";

export class UploadHandler extends AbstractHandler {
  private readonly MAX_FILE_SIZE = 500 * 1024 * 1024; // 500MB
  private readonly PACKAGES_PREFIX = "packages/";

  constructor(private config: Config) {
    super();
  }

  canHandle(request: Request): boolean {
    const url = new URL(request.url);
    return url.pathname === "/api/upload" && request.method === "POST";
  }

  async handle(request: Request, env: Env, ctx: ExecutionContext): Promise<Response> {
    // 1. 验证 API Key
    const apiKey = request.headers.get("X-API-Key");
    if (!this.validateApiKey(apiKey, env)) {
      return this.json(
        { error: { code: "UNAUTHORIZED", message: "Invalid API Key" } },
        { status: 401 }
      );
    }

    // 2. 解析表单数据
    const formData = await request.formData();
    const file = formData.get("spk") as File | null;

    if (!file) {
      return this.json(
        { error: { code: "NO_FILE", message: "No SPK file provided" } },
        { status: 400 }
      );
    }

    // 3. 验证文件
    const validationError = this.validateFile(file);
    if (validationError) {
      return this.json(validationError, { status: 400 });
    }

    // 4. 上传到 R2
    const r2Key = `${this.PACKAGES_PREFIX}${file.name}`;
    await env.SPKS_BUCKET.put(r2Key, file.stream(), {
      httpMetadata: { contentType: "application/octet-stream" }
    });

    // 5. 解析并缓存元数据
    const content = await file.arrayBuffer();
    const metadata = await this.parseAndCache(env, r2Key, content);

    return Response.json({
      success: true,
      filename: file.name,
      package: metadata.package,
      version: metadata.version,
      arch: metadata.arch,
      message: "Package uploaded and indexed"
    });
  }

  private validateApiKey(apiKey: string | null, env: Env): boolean {
    return apiKey === env.SSPKS_API_KEY;
  }

  private validateFile(file: File): UploadError | null {
    if (file.size > this.MAX_FILE_SIZE) {
      return { error: { code: "FILE_TOO_LARGE", message: "File exceeds limit" } };
    }
    if (!file.name.toLowerCase().endsWith(".spk")) {
      return { error: { code: "INVALID_FILE_TYPE", message: "Only .spk files allowed" } };
    }
    return null;
  }

  private async parseAndCache(
    env: Env,
    r2Key: string,
    content: ArrayBuffer
  ): Promise<{ package: string; version: string; arch: string[] }> {
    const info = await Package.extractInfo(content);
    if (!info) {
      return { package: r2Key.replace(".spk", ""), version: "unknown", arch: [] };
    }

    await env.SPKS_CACHE.put(`pkg:${info.package}`, JSON.stringify({ key: r2Key, metadata: info }), {
      expirationTtl: 3600
    });

    return { package: info.package || "", version: info.version || "", arch: info.arch || [] };
  }
}
```

```typescript
import { cloudflareTest } from "@cloudflare/vitest-pool-workers";

export default defineConfig({

    cloudflareTest({
      wrangler: { configPath: "./wrangler.toml" },
```

```bash
# 运行所有测试
npm test

# 监听模式

# 覆盖率报告
npm run test:coverage
```

```

```

### 测试示例

```typescript
import { describe, it, expect, vi } from 'vitest';
import { PackageFilter } from '../src/package/PackageFilter';

describe('PackageFilter', () => {
  it('should filter by architecture', () => {
    const filter = new PackageFilter(mockPackages);
    filter.setArchitectureFilter('avoton');

    const result = filter.getFilteredPackageList();

    expect(result.every(p => p.isCompatibleToArch('avoton'))).toBe(true);
  });

  it('should return latest version only', () => {
    const filter = new PackageFilter(mockPackages);
    filter.setOldVersionFilter(true);

    const result = filter.getFilteredPackageList();

    // 验证没有重复包名
    const packageNames = result.map(p => p.metadata.package);
    expect(new Set(packageNames).size).toBe(packageNames.length);
  });
});
```

## 性能优化

### 1. 缓存策略

```typescript
// API 响应缓存
const cacheKey = `packages:${arch}:${channel}`;
const cached = await env.SPKS_CACHE.get(cacheKey);
if (cached) {
  return new Response(cached);
}

// 缓存 5 分钟
response.headers.set('Cache-Control', 'public, max-age=300');
```

### 2. 避免冷启动

```typescript
// 预热 KV 缓存
async function warmupCache(env: Bindings) {
  const packages = await listPackages(env.SPKS_BUCKET);
  for (const pkg of packages) {
    await env.SPKS_CACHE.put(`package:${pkg.name}`, JSON.stringify(pkg));
  }
}
```

### 3. 减少 R2 调用

```typescript
// 批量获取对象

  prefix: 'packages/',

});
```
